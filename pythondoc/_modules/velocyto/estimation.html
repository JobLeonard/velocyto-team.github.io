def _fit1_slope(y: np.ndarray, x: np.ndarray) -> float:
    """Simple function that fit a linear regression model without intercept
    """
    if not np.any(x):
        m = np.NAN  # It is definetelly not at steady state!!!
    elif not np.any(y):
        m = 0
    else:
        result, rnorm = scipy.optimize.nnls(x[:, None], y)  # Fastest but costrains result >= 0
        m = result[0]
        # Second fastest: m, _ = scipy.optimize.leastsq(lambda m: x*m - y, x0=(0,))
        # Third fastest: m = scipy.optimize.minimize_scalar(lambda m: np.sum((x*m - y)**2 )).x
        # Before I was doinf fastest: scipy.optimize.minimize_scalar(lambda m: np.sum((y - m * x)**2), bounds=(0, 3), method="bounded").x
        # Optionally one could clip m if high value make no sense
        # m = np.clip(m,0,3)
    return m


def _fit1_slope_weighted(y: np.ndarray, x: np.ndarray, w: np.ndarray, bounds: Tuple[float, float]=(0, 3)) -> float:
    """Simple function that fit a weighted linear regression model without intercept
    """
    if not np.any(x):
        m = np.NAN  # It is definetelly not at steady state!!!
    elif not np.any(y):
        m = 0
    else:
        m = scipy.optimize.minimize_scalar(lambda m: np.sum(w * (x * m - y)**2), bounds=(0, 3), method="bounded").x
    return m


def _fit1_slope_weighted_offset(y: np.ndarray, x: np.ndarray, w: np.ndarray, fixperc_q: bool=False, limit_gamma: bool=False) -> Tuple[float, float]:
    """Function that fits a weighted linear regression model with intercept
    with some adhoc
    """
    if not np.any(x):
        m = (np.NAN, 0)  # It is definetelly not at steady state!!!
    elif not np.any(y):
        m = (0, 0)
    else:
        if fixperc_q:
            m1 = np.percentile(y[x <= np.percentile(x, 1)], 50)
            m0 = scipy.optimize.minimize_scalar(lambda m: np.sum(w * (x * m - y + m1)**2), bounds=(0, 3), method="bounded").x
            m = (m0, m1)
        else:
            # m, _ = scipy.optimize.leastsq(lambda m: np.sqrt(w) * (-y + x * m[0] + m[1]), x0=(0, 0))  # This is probably faster but it can have negative slope
            # NOTE: The up_gamma is to deal with cases where consistently y > x. Those should have positive velocity everywhere
            if limit_gamma:
                if np.median(y) > np.median(x):
                    high_x = x > np.percentile(x, 90)
                    up_gamma = np.percentile(y[high_x], 10) / np.median(x[high_x])
                    up_gamma = np.maximum(1.5, up_gamma)
                else:
                    up_gamma = 1.5  # Just a bit more than 1
            else:
                up_gamma = 30
            up_q = 2 * np.sum(y * w) / np.sum(w)
            m = scipy.optimize.minimize(lambda m: np.sum(w * (-y + x * m[0] + m[1])**2),
                                        x0=(0.1, 1e-16), method="L-BFGS-B",
                                        bounds=[(1e-8, up_gamma), (0, up_q)]).x  # If speedup is needed either the gradient or numexpr could be used
    return m[0], m[1]

[docs]def fit_slope_weighted_offset(Y: np.ndarray, X: np.ndarray, W: np.ndarray, fixperc_q: bool=False, return_R2: bool=True, limit_gamma: bool=False) -> Any:
    """Loop through the genes and fits the slope

    Y: np.ndarray, shape=(genes, cells)
        the dependent variable (unspliced)
    X: np.ndarray, shape=(genes, cells)
        the independent variable (spliced)
    """
    # NOTE this could be easily parallelized
    slopes = np.zeros(Y.shape[0], dtype="float32")
    offsets = np.zeros(Y.shape[0], dtype="float32")
    if return_R2:
        R2 = np.zeros(Y.shape[0], dtype="float32")
    for i in range(Y.shape[0]):
        m, q = _fit1_slope_weighted_offset(Y[i, :], X[i, :], W[i, :], fixperc_q, limit_gamma)
        slopes[i] = m
        offsets[i] = q
        if return_R2:
            # NOTE: the coefficient of determination is not weighted but the fit is
            SSres = np.sum((m * X[i, :] + q - Y[i, :])**2)
            SStot = np.sum((Y[i, :].mean() - Y[i, :])**2)
            R2[i] = 1 - (SSres / SStot)
    if return_R2:
        return slopes, offsets, R2
    return slopes, offsets




[docs]def naive_slope(y: np.ndarray, x: np.ndarray) -> float:
    """Slope calculate by the ratio of 99th percentiles
    
        Arguments
        ---------
        y: np.ndarray  shape= (cells,)
        
        x: np.ndarray  shape= (cells,)

        Returns
        -------
        The slope
    """
    yp = np.percentile(y, q=99.5, axis=1)
    xp = np.percentile(x, q=99.5, axis=1)
    if xp == 0:
        return np.nan
    if yp == 0:
        return 0
    return yp / xp

    """Neighbour wise fit of gamma with multiple qs
    input shoud be (cells, genes) sic!

    Returns
    -------
    gammas: np.ndarray = (genes)

    qs: np.ndarray, shape = (neighborhoods, genes)

    """
    
    n_genes = S.shape[1]
    ui_labels = np.unique(labels)
    n_neighbourhoods = len(ui_labels)
    assert np.allclose(ui_labels, np.arange(np.max(labels) + 1)), "All from 0 to max labels should be present"

    def objective_fun(x: np.ndarray, U: np.ndarray, S: np.ndarray, labels: np.ndarray) -> None:
        gamma, q = x[:n_genes], x[n_genes:].reshape((-1, n_genes))
        epsilon_ij = U - S * gamma - q[labels, :]
        # TODO Add regularization
        return np.sum(epsilon_ij**2)

    logging.debug("Prepare the initialization values and the bounds for L-BFGS-B")
    gammmas0 = fit_slope(U.T, S.T)
    qs0 = 1e-6 * np.ones((n_neighbourhoods, n_genes))
    x0 = np.hstack([gammmas0, qs0.flat[:]])
    bounds_gammas = [(0, 4) for i in range(n_genes)]
    bounds_qs = [np.maximum(np.mean(U[labels == r, c]), 0.01) for r in range(n_neighbourhoods) for c in range(n_genes)]
    if q_positive:
        bounds_qs = [(0, i) for i in bounds_qs]
    else:
        bounds_qs = [(-i, i) for i in bounds_qs]
    logging.debug("Running L-BFGS-B")

    class IterCounter:
        def __init__(self) -> None:
            self.global_counter = 0

        def callback_fun(self, xk: np.ndarray) -> None:
            if self.global_counter % 20 == 0:
                logging.debug(f"Iteration #{self.global_counter+1}")
            self.global_counter += 1
    
    res = scipy.optimize.minimize(objective_fun, x0=x0, args=(U, S, labels.astype("int32")), method="L-BFGS-B",
                                  bounds=bounds_gammas + bounds_qs, callback=IterCounter().callback_fun)

    return res.x[:n_genes], res.x[n_genes:].reshape((-1, n_genes))

    """Neighbour wise fit of gamma with multiple qs and regularization using the neighborhoods q smoothing
    input should be (cells, genes) sic!

    Arguments
    ---------
    U: np.ndarray (cells, genes)
        Unspliced molecules table or correspondent non negative components
    S: np.ndarray (cells, genes)
        Spliced molecules table or correspondent non negative components
    CS: np.ndarray, shape=(n_centroids, genes)
        Centroids coordinates
    labels: np.ndarray
        labels of the neighborhood, must be dense for 0 to n
    graph_design: sparse.csr_matrix
        a sparse matrix with weights used in the regularization as values
    alpha: float
        how strongly the regularization weights over the least squares
    kind_objective: str
        "project_q": projects q on the vector that contains the centroid and
                     then impose that q is similar to the mean of the projected vectors
        "neighborhood": then impose that q is similar to the mean of the neighbours
        In both cases the weights are determined by the input graph_design
    q_positive: bool, default=False
        Wether to constrain q to positive values

    Returns
    -------
    gammas: np.ndarray = (genes, neighborhoods)

    qs: np.ndarray, shape = (genes, neighborhoods)

    """
    
    n_genes = S.shape[1]
    n_neighbourhoods = len(np.unique(labels))

    if kind_objective == "neighborhood":

        def objective_fun(x: np.ndarray, U: np.ndarray, S: np.ndarray, CS: np.ndarray,
                          labels: np.ndarray, w: np.ndarray, alpha: float) -> None:
            n_genes = S.shape[1]
            gamma, q = x[:n_genes], x[n_genes:].reshape((-1, n_genes))
            epsilon_ij = U - S * gamma - q[labels, :]
            regularizer = np.sum((w @ q)**2)  # np.sum((q - (w @ q))**2)
            return np.sum(epsilon_ij**2) + alpha * regularizer

    elif kind_objective == "project_q":

        def objective_fun(x: np.ndarray, U: np.ndarray, S: np.ndarray, CS: np.ndarray,
                          labels: np.ndarray, w: np.ndarray, alpha: float) -> None:
            n_genes = S.shape[1]
            gamma, q = x[:n_genes], x[n_genes:].reshape((-1, n_genes))
            epsilon_ij = U - S * gamma - q[labels, :]
            regularizer = regularizer_projecting(CS, q, w)
            return np.sum(epsilon_ij**2) + alpha * regularizer
    else:
        raise ValueError("kind_objective specified does not exist")

    logging.debug("Prepare the initialization values and the bounds for L-BFGS-B")
    gammmas0 = fit_slope(U.T, S.T)
    qs0 = 1e-6 * np.ones((n_neighbourhoods, n_genes))
    x0 = np.hstack([gammmas0, qs0.flat[:]])
    bounds_gammas = [(0, 4) for i in range(n_genes)]
    bounds_qs = [np.maximum(np.mean(U[labels == r, c]), 0.01) for r in range(n_neighbourhoods) for c in range(n_genes)]
    if q_positive:
        bounds_qs = [(0, i) for i in bounds_qs]
    else:
        bounds_qs = [(-i, i) for i in bounds_qs]
    logging.debug("Running L-BFGS-B")
    
    class IterCounter:
        def __init__(self) -> None:
            self.global_counter = 0

        def callback_fun(self, xk: np.ndarray) -> None:
            if self.global_counter % 5 == 0:
                logging.debug(f"Iteration #{self.global_counter+1}")
                logging.debug(f"Objective is: {xk:.6e}")
            self.global_counter += 1
    
    res = scipy.optimize.minimize(objective_fun, x0=x0, args=(U, S, CS, labels.astype("int32"), graph_design.A, alpha), method="L-BFGS-B",

[docs]def estimate_velocity(U: np.ndarray, S: np.ndarray, steadystate: np.ndarray) -> np.ndarray:
    """Estimate velocity for all the genes using gene-relative model (numpy version)

    Arguments
    ---------
    U : np.ndarray (genes, cells)
        The unspliced molecule count
    S : np.ndarray (genes, cells)
        The spliced molecule count
    steadystate : np.ndarray(bool), shape=cells
        A boolean array, True when the cell is a steady state cell

    Returns
    -------
    velcity: np.ndarray (shape=S.shape)
        The velocity for every gene.

    NOTE:
    Special caseses:
        if not np.any(S[gene, steadystate]): delta = NaN
        if not np.any(U[gene, steadystate]): delta = 0

    """

    S_steady = S[:, steadystate]
    U_steady = U[:, steadystate]
    gamma = fit_slope(U_steady, S_steady)  # degradation rate
    gamma[np.isinf(gamma)] = np.nan  # Put a nan so that it just propagates
    v = U - S * gamma[:, None]
    v[np.isnan(v)] = 0  # All the nans go to 0
    return v




[docs]def clusters_stats(U: np.ndarray, S: np.ndarray,
                   clusters_uid: np.ndarray, cluster_ix: np.ndarray,
                   size_limit: int=40) -> Tuple[np.ndarray, np.ndarray]:
    """Calculate the averages per cluster
    
    If the cluster is too small (size<size_limit) the average of the toal is reported instead
    """
    U_avgs = np.zeros((S.shape[0], len(clusters_uid)))
    S_avgs = np.zeros((S.shape[0], len(clusters_uid)))
    avgU_div_avgS = np.zeros((S.shape[0], len(clusters_uid)))
    slopes_by_clust = np.zeros((S.shape[0], len(clusters_uid)))
    for i, uid in enumerate(clusters_uid):
        cluster_filter = cluster_ix == i
        n_cells = np.sum(cluster_filter)
        logging.info(f"Cluster: {uid} ({n_cells} cells)")
        if n_cells > size_limit:
            U_avgs[:, i], S_avgs[:, i] = U[:, cluster_filter].mean(1), S[:, cluster_filter].mean(1)
        else:
            U_avgs[:, i], S_avgs[:, i] = U.mean(1), S.mean(1)
            
    return U_avgs, S_avgs
